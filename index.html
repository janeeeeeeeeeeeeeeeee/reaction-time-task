<script>

    const CONFIG = {

        ENDPOINT: 'YOUR_WEB_APP_URL', // 必须替换为有效URL

        TRIALS: 10,

        RADIUS: 50,

        DELAY_MS: 500,

        BIN_WIDTH: 30 // 分箱精度 (ms)

    };


    let state = {

        id: null,

        trial: 0,

        t0: 0,

        x: -1,

        y: -1,

        payloads: []

    };


    const dom = {

        init: document.getElementById('screen-init'),

        task: document.getElementById('screen-task'),

        result: document.getElementById('screen-result'),

        input: document.getElementById('p-id'),

        btn: document.getElementById('btn-exec'),

        cvs: document.getElementById('canvas-core'),

        chart: document.getElementById('chart-core'),

        status: document.getElementById('status-text')

    };


    const ctx = dom.cvs.getContext('2d');


    const syncViewport = () => {

        dom.cvs.width = window.innerWidth;

        dom.cvs.height = window.innerHeight;

    };

    window.addEventListener('resize', syncViewport);

    syncViewport();


    dom.btn.addEventListener('click', () => {

        const rawId = dom.input.value.trim();

        state.id = rawId || Math.random().toString(36).substring(2, 10).toUpperCase();

        

        dom.init.style.display = 'none';

        dom.task.style.display = 'block';

        

        setTimeout(cycle, CONFIG.DELAY_MS);

    });


    const cycle = () => {

        if (state.trial >= CONFIG.TRIALS) {

            terminate();

            return;

        }

        

        state.trial++;

        

        const w = dom.cvs.width;

        const h = dom.cvs.height;

        const r = CONFIG.RADIUS;

        

        state.x = r + Math.random() * (w - r * 2);

        state.y = r + Math.random() * (h - r * 2);

        

        ctx.clearRect(0, 0, w, h);

        ctx.fillStyle = '#fff';

        ctx.beginPath();

        ctx.arc(state.x, state.y, r, 0, Math.PI * 2);

        ctx.fill();

        

        state.t0 = performance.now();

    };


    dom.cvs.addEventListener('mousedown', (e) => {

        if (state.trial === 0 || state.trial > CONFIG.TRIALS) return;

        

        const dx = e.clientX - state.x;

        const dy = e.clientY - state.y;

        const dist = Math.sqrt(dx * dx + dy * dy);

        

        if (dist <= CONFIG.RADIUS) {

            const rt = performance.now() - state.t0;

            

            state.payloads.push({

                Participant_ID: state.id,

                Trial_Number: state.trial,

                Reaction_Time: parseFloat(rt.toFixed(2))

            });

            

            ctx.clearRect(0, 0, dom.cvs.width, dom.cvs.height);

            state.x = -1;

            state.y = -1;

            

            setTimeout(cycle, CONFIG.DELAY_MS);

        }

    });


    const terminate = async () => {

        dom.task.style.display = 'none';

        dom.result.style.display = 'flex';


        const rtSum = state.payloads.reduce((acc, curr) => acc + curr.Reaction_Time, 0);

        const userAvg = rtSum / CONFIG.TRIALS;


        // 异步写入协议

        state.payloads.forEach(data => {

            fetch(CONFIG.ENDPOINT, {

                method: 'POST',

                mode: 'no-cors',

                headers: { 'Content-Type': 'text/plain' },

                body: JSON.stringify(data)

            }).catch(() => {});

        });


        // 提取历史向量

        try {

            dom.status.innerText = 'FETCHING_HISTORICAL_DISTRIBUTION...';

            const response = await fetch(CONFIG.ENDPOINT);

            

            if (!response.ok) throw new Error('NETWORK_ERR');

            

            const rawData = await response.json();

            const validHistory = rawData

                .map(Number)

                .filter(n => !isNaN(n) && n > 50 && n < 5000); 


            // 将当前个体的均值作为张量引入

            validHistory.push(userAvg);

            

            dom.status.innerText = `SYS_SYNC | N=${validHistory.length} | ID:${state.id}`;

            renderHistogram(validHistory, userAvg);


        } catch (err) {

            dom.status.innerText = 'ERR_READ_PHASE: ' + err.message;

            dom.status.style.color = '#500';

            // 降级渲染（仅含当前样本）

            renderHistogram([userAvg], userAvg);

        }

    };


    const renderHistogram = (data, thresholdX) => {

        const w = CONFIG.BIN_WIDTH;

        const min = Math.floor(Math.min(...data) / w) * w;

        const max = Math.ceil(Math.max(...data) / w) * w;

        

        const labels = [];

        const bins = [];

        

        for (let i = min; i <= max; i += w) {

            labels.push(i);

            bins.push(0);

        }

        

        data.forEach(val => {

            const index = Math.floor((val - min) / w);

            if (index >= 0 && index < bins.length) {

                bins[index]++;

            } else if (index === bins.length) {

                bins[index - 1]++;

            }

        });


        const ctxChart = dom.chart.getContext('2d');

        

        new Chart(ctxChart, {

            type: 'bar',

            data: {

                labels: labels,

                datasets: [{

                    label: 'Δ Frequency',

                    data: bins,

                    backgroundColor: '#111',

                    borderColor: '#222',

                    borderWidth: 1,

                    barPercentage: 1.0,

                    categoryPercentage: 1.0

                }]

            },

            options: {

                responsive: true,

                maintainAspectRatio: false,

                animation: { duration: 800, easing: 'easeOutQuart' },

                scales: {

                    x: { 

                        grid: { color: '#0a0a0a' }, 

                        ticks: { color: '#444', maxTicksLimit: 20 } 

                    },

                    y: { 

                        grid: { color: '#0a0a0a' }, 

                        ticks: { color: '#444', stepSize: 1 }, 

                        beginAtZero: true 

                    }

                },

                plugins: {

                    legend: { display: false },

                    tooltip: { enabled: false }

                }

            },

            plugins: [{

                id: 'structuralTensionLine',

                afterDatasetsDraw: (chart) => {

                    const { ctx, chartArea: { top, bottom }, scales: { x } } = chart;

                    

                    // 悬停/错位：线性插值映射离散轴

                    const relativeOffset = (thresholdX - min) / w;

                    

                    let targetPx;

                    if (relativeOffset <= 0) targetPx = x.getPixelForTick(0);

                    else if (relativeOffset >= labels.length - 1) targetPx = x.getPixelForTick(labels.length - 1);

                    else {

                        const idx = Math.floor(relativeOffset);

                        const remainder = relativeOffset - idx;

                        const p1 = x.getPixelForTick(idx);

                        const p2 = x.getPixelForTick(idx + 1);

                        targetPx = p1 + (p2 - p1) * remainder;

                    }


                    ctx.save();

                    ctx.beginPath();

                    ctx.moveTo(targetPx, top);

                    ctx.lineTo(targetPx, bottom);

                    ctx.lineWidth = 1;

                    ctx.strokeStyle = '#888';

                    ctx.setLineDash([4, 4]);

                    ctx.stroke();

                    

                    ctx.fillStyle = '#aaa';

                    ctx.font = '11px monospace';

                    ctx.fillText(`μ=${thresholdX.toFixed(1)}`, targetPx + 6, top + 12);

                    ctx.restore();

                }

            }]

        });

    };

</script>
